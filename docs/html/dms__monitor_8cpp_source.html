21cv::Rect DMSMonitor::detectFace(const cv::Mat& frame) {
22 // Защита: если сеть не загружена или кадр пустой
23 if (face_net.empty() || frame.empty()) return cv::Rect();
24
25 try {
26 cv::Mat blob = cv::dnn::blobFromImage(frame, 1.0, cv::Size(300, 300), cv::Scalar(104, 177, 123));
27 face_net.setInput(blob);
28 cv::Mat detection = face_net.forward();
29 cv::Mat detectionMat(detection.size[2], detection.size[3], CV_32F, detection.ptr<float>());
30
31 for (int i = 0; i < detectionMat.rows; i++) {
32 float confidence = detectionMat.at<float>(i, 2);
33 if (confidence > 0.5) {
34 int x1 = static_cast<int>(detectionMat.at<float>(i, 3) * frame.cols);
35 int y1 = static_cast<int>(detectionMat.at<float>(i, 4) * frame.rows);
36 int x2 = static_cast<int>(detectionMat.at<float>(i, 5) * frame.cols);
37 int y2 = static_cast<int>(detectionMat.at<float>(i, 6) * frame.rows);
38 return cv::Rect(cv::Point(x1, y1), cv::Point(x2, y2)) & cv::Rect(0, 0, frame.cols, frame.rows);
39 }
40 }
41 } catch (...) {
42 return cv::Rect();
43 }
44 return cv::Rect();

75DriverState DMSMonitor::analyze(cv::Mat& frame) {
76 DriverState state;
77
78 // Если кадр пустой, просто возвращаем пустой стейт (Шаг 7.6)
79 if (frame.empty()) {
80 state.face_detected = false;
81 return state;
82 }
83
84 state.face_rect = detectFace(frame);
85
86 if (!state.face_rect.empty()) {
87 state.face_detected = true;
88 state.eyes_open = estimateEyeOpenness(frame, state.face_rect);
89 state.head_turn_deg = estimateHeadTurn(state.face_rect, frame.cols);
90 state.looking_forward = std::abs(state.head_turn_deg) < 15.0f;
91 state.alert_distracted = !state.looking_forward;
92 }
93
94 // История обновляется только если есть кадр
95 eye_history.push_back(state.eyes_open);
96 if (eye_history.size() > HISTORY_SIZE) eye_history.pop_front();
97
98 int closed_count = 0;
99 for (bool open : eye_history) if (!open) closed_count++;
100
101 if (closed_count >= DROWSY_THRESHOLD && state.face_detected) {
102 state.alert_drowsy = true;
103 }
104
105 return state;
106}
