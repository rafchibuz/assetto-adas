20void obdThreadFunc(SharedState& state, OBDParser& parser, ONNXClassifier& classifier) {
21 int record_idx = 0;
22 int total = parser.getCount();
23
24 while (state.running && record_idx < total) {
25 OBDRecord rec = parser.getRecord(record_idx);
26
27 // Инференс нейросети
28 std::vector<float> features = {
29 (float)rec.time, (float)rec.speed, (float)rec.rpm,
30 (float)rec.throttle, (float)rec.accel_lon, (float)rec.accel_lat
31 };
32 ClassificationResult res = classifier.classify(features);
33
34 // Синхронизация и обновление SharedState
35 {
36 std::lock_guard<std::mutex> lock(state.mtx);
37 state.current_record = rec;
38 state.current_style = res.label;
39 state.confidence = res.confidence;
40 state.total_obd_processed++;
41
42 if (res.label == 2) state.alert_aggressive_count++;
43 }
44
45 record_idx++;
46 std::this_thread::sleep_for(std::chrono::milliseconds(100)); // 10 Hz
47 }
48 state.running = false;
49}
