49std::vector<float> ONNXClassifier::softmax(const std::vector<float>& logits) {
50 std::vector<float> exp_logits(logits.size());
51 float max_logit = *std::max_element(logits.begin(), logits.end());
52 float sum = 0.0f;
53 for (size_t i = 0; i < logits.size(); ++i) {
54 exp_logits[i] = std::exp(logits[i] - max_logit);
55 sum += exp_logits[i];
56 }
57 for (float& val : exp_logits) val /= sum;
58 return exp_logits;
59}
